tracerate

A no-nonsense CLI internet speed tester.

bash 
# download
curl -fsSL https://executables.devjugal.com/tracerate/latest/tracerate-linux-amd64 -o tracerate

# verify checksum
curl -fsSL https://executables.devjugal.com/tracerate/latest/tracerate-linux-amd64.sha256 \
  | sha256sum -c

# install
chmod +x tracerate
sudo mv tracerate /usr/local/bin/tracerate

# run
tracerate

powershell 
# download
Invoke-WebRequest -Uri "https://executables.devjugal.com/tracerate/latest/tracerate-windows-amd64.exe" `
  -OutFile tracerate.exe

# verify checksum
$expected = (Invoke-WebRequest -Uri "https://executables.devjugal.com/tracerate/latest/tracerate-windows-amd64.exe.sha256").Content.Trim().Split()[0]
$actual = (Get-FileHash tracerate.exe -Algorithm SHA256).Hash.ToLower()
if ($expected -ne $actual) { throw "checksum mismatch" }

# install
Move-Item tracerate.exe "$env:USERPROFILE\AppData\Local\Microsoft\WindowsApps\tracerate.exe"

# run
tracerate
